```kotlin
@Composable
fun ModularHomeScreen(
    onModuleClick: (ModuleInfo) -> Unit,
    onQuickActionClick: (QuickAction) -> Unit
) {
    Scaffold(
        topBar = {
            HomeTopBar()
        }
    ) { paddingValues ->
        LazyColumn(
            modifier = Modifier
                .fillMaxSize()
                .padding(paddingValues),
            contentPadding = PaddingValues(16.dp),
            verticalArrangement = Arrangement.spacedBy(16.dp)
        ) {
            // 欢迎区域
            item {
                WelcomeSection()
            }
            
            // 数据概览卡片
            item {
                DataOverviewCard()
            }
            
            // 常用功能模块
            item {
                FavoriteModulesSection(
                    modules = getFavoriteModules(),
                    onModuleClick = onModuleClick
                )
            }
            
            // 所有模块网格
            item {
                AllModulesSection(
                    modules = getAllModules(),
                    onModuleClick = onModuleClick,
                    onEditClick = { /* 进入编辑模式 */ }
                )
            }
        }
    }
}
```

## 2. 模块卡片设计

```kotlin
@Composable
fun ModuleCard(
    module: ModuleInfo,
    onClick: () -> Unit,
    isEditMode: Boolean = false,
    modifier: Modifier = Modifier
) {
    Card(
        modifier = modifier
            .size(100.dp)
            .clip(RoundedCornerShape(16.dp))
            .clickable { onClick() },
        colors = CardDefaults.cardColors(
            containerColor = module.backgroundColor.copy(alpha = 0.1f)
        )
    ) {
        Box(
            modifier = Modifier.fillMaxSize(),
            contentAlignment = Alignment.Center
        ) {
            Column(
                horizontalAlignment = Alignment.CenterHorizontally,
                verticalArrangement = Arrangement.Center
            ) {
                // 模块图标
                Box(
                    modifier = Modifier
                        .size(48.dp)
                        .clip(CircleShape)
                        .background(module.backgroundColor),
                    contentAlignment = Alignment.Center
                ) {
                    Icon(
                        imageVector = module.icon,
                        contentDescription = module.name,
                        tint = Color.White,
                        modifier = Modifier.size(28.dp)
                    )
                }
                
                Spacer(modifier = Modifier.height(8.dp))
                
                // 模块名称
                Text(
                    text = module.name,
                    style = MaterialTheme.typography.bodyMedium,
                    maxLines = 1,
                    overflow = TextOverflow.Ellipsis
                )
                
                // 角标（如有新消息）
                if (module.badgeCount > 0) {
                    Badge(
                        modifier = Modifier
                            .align(Alignment.TopEnd)
                            .padding(4.dp)
                    ) {
                        Text(module.badgeCount.toString())
                    }
                }
                
                // 编辑模式下的删除按钮
                if (isEditMode) {
                    IconButton(
                        onClick = { /* 删除/隐藏模块 */ },
                        modifier = Modifier
                            .align(Alignment.TopEnd)
                            .size(24.dp)
                    ) {
                        Icon(
                            Icons.Default.Close,
                            contentDescription = "删除",
                            tint = MaterialTheme.colorScheme.error
                        )
                    }
                }
            }
        }
    }
}
```

## 4. 快捷操作按钮（FAB）

```kotlin
@Composable
fun QuickActionFAB(
    isExpanded: Boolean,
    onToggle: () -> Unit,
    onActionClick: (QuickAction) -> Unit
) {
    Box(
        modifier = Modifier.fillMaxSize(),
        contentAlignment = Alignment.BottomEnd
    ) {
        // 背景遮罩
        if (isExpanded) {
            Box(
                modifier = Modifier
                    .fillMaxSize()
                    .background(Color.Black.copy(alpha = 0.5f))
                    .clickable { onToggle() }
            )
        }
        
        Column(
            horizontalAlignment = Alignment.End,
            verticalArrangement = Arrangement.spacedBy(16.dp),
            modifier = Modifier.padding(16.dp)
        ) {
            // 快捷操作选项
            AnimatedVisibility(visible = isExpanded) {
                Column(
                    verticalArrangement = Arrangement.spacedBy(12.dp)
                ) {
                    QuickActionItem(
                        icon = Icons.Default.Add,
                        label = "快速记账",
                        backgroundColor = Color(0xFF4CAF50),
                        onClick = { onActionClick(QuickAction.ADD_TRANSACTION) }
                    )
                    
                    QuickActionItem(
                        icon = Icons.Default.Task,
                        label = "添加待办",
                        backgroundColor = Color(0xFF2196F3),
                        onClick = { onActionClick(QuickAction.ADD_TODO) }
                    )
                    
                    QuickActionItem(
                        icon = Icons.Default.CheckCircle,
                        label = "习惯打卡",
                        backgroundColor = Color(0xFFFF9800),
                        onClick = { onActionClick(QuickAction.HABIT_CHECK) }
                    )
                }
            }
            
            // 主FAB按钮
            FloatingActionButton(
                onClick = onToggle,
                containerColor = MaterialTheme.colorScheme.primary
            ) {
                Icon(
                    imageVector = if (isExpanded) Icons.Default.Close else Icons.Default.Add,
                    contentDescription = "快捷操作",
                    modifier = Modifier.animateContentSize()
                )
            }
        }
    }
}
```

## 6. 新的底部导航栏

```kotlin
@Composable
fun ModularBottomNavBar(
    navController: NavController,
    showQuickAction: Boolean = true
) {
    val navBackStackEntry by navController.currentBackStackEntryAsState()
    val currentDestination = navBackStackEntry?.destination
    
    NavigationBar {
        // 首页
        NavigationBarItem(
            icon = {
                Icon(
                    imageVector = if (currentDestination?.route == "home") 
                        Icons.Filled.Home else Icons.Outlined.Home,
                    contentDescription = "首页"
                )
            },
            label = { Text("首页") },
            selected = currentDestination?.route == "home",
            onClick = {
                navController.navigate("home") {
                    popUpTo(navController.graph.findStartDestination().id) {
                        saveState = true
                    }
                    launchSingleTop = true
                    restoreState = true
                }
            }
        )
        
        // 占位符（给FAB留空间）
        if (showQuickAction) {
            NavigationBarItem(
                icon = { },
                label = { },
                selected = false,
                onClick = { },
                enabled = false
            )
        }
        
        // 我的
        NavigationBarItem(
            icon = {
                Icon(
                    imageVector = if (currentDestination?.route == "profile") 
                        Icons.Filled.Person else Icons.Outlined.Person,
                    contentDescription = "我的"
                )
            },
            label = { Text("我的") },
            selected = currentDestination?.route == "profile",
            onClick = {
                navController.navigate("profile") {
                    popUpTo(navController.graph.findStartDestination().id) {
                        saveState = true
                    }
                    launchSingleTop = true
                    restoreState = true
                }
            }
        )
    }
}
```

## 效果预览

### 优点
1. **简洁清爽**：底部只有2-3个导航项
2. **扩展性强**：新模块直接添加到网格中
3. **个性化**：支持自定义模块排序和显示
4. **操作便捷**：快捷操作一键直达

### 交互流程
1. 用户打开应用，看到简洁的首页
2. 通过模块卡片进入具体功能
3. 使用FAB快速创建内容
4. 在"我的"页面管理个人设置和模块配置
